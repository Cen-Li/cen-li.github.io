void Exchange(int & number1, int & number2);

int main()
{
	int  num1=3, num2=5;

	count<<num1<<"\t"<<num2<<endl;

	Exchange(num1, num2);

	count<<num1<<"\t"<<num2<<endl;

	return 0;
}

void Exchange(int & number1, int & number2)
{
	int temp;

	temp = number1;
	number1 = number2;
	number2 = temp;

	return;
}

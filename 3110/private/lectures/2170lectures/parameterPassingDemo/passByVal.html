#include <iostream>
#include <string>
using namespace std;
void passByVal(int val);

int main()
{
	int num;
	cout <<"please enter an integer value: ";
	cin >> num;

	cout << "num = " << num << endl;
	passByVal(num);
	cout << "num = " << num << endl;
	return 0;
}

void passByVal(int number)
{
	number = number*2;
	cout << "number  = " << number << endl;
}
